JavaScript Function Concepts

1] What is a Function?

  1. A function is block of reusable code that performs a task.
  2. Follow DRY principle -> Don't Repeat Yourself.
  3. Can be called multiple times with different inputs.
            function run1()
            {
                function  msg()
                {
                    return "Here is Simple Message";
                }
                document.getElementById("out1").textContent = msg();
                console.log(msg());
            }

2] Fuction Declaration vs Function Expression

  1. Declaration: Hoisted -> can be called before defined
  2. Expression: Stored in variable, not hoisted
            function run2()
            {
                function  addDec(a,b) 
                {
                    return a+b;
                }
                const addExp = function(a, b) 
                {
                    return a + b;
                };
                let result = "Declaration: "+addDec(5,3)+"|Expression: "+addExp(5,3);

                document.getElementById("out2").textContent = result;
                console.log(result);
            }

3] Parameter Vs Arguments

  1. Parameters: Variables listed in function definition.
  2. Arguments: Actual Values passed during function call.
            function run3()
            {
                function  mul(x1, x2)
                {
                    return x1*x2;
                }
                let m1 = mul(12,5);

                document.getElementById("out3").textContent = "multiply Result: "+m1;
                console.log("Multiply result "+ m1);
            }

4] Return Statements

  1. return sends value back to caller.
  2. Without return, function gives undefined.
            function run4()
            {
                function  square(num)
                {
                    return num * num;
                }
                let result = square(7);

                document.getElementById("out4").textContent = "Square of : 7 = "+result;
                console.log("Square of : 6 = " +result);
            }

5] Arrow Function (ES6)

  1. Introduced in ES6 for clearer syntax.
  2. Shorter than traditional functions.
  3. Usedful for callbacks and small tasks
            function run5()
            {
                const greet = (name) => `Hello, ${name}!`;
                const square = x => x * x;

                let result = greet("Akshay") + " | Square of 5: " +square(5);

                document.getElementById("out5").textContent = result;
                console.log(result);
            }

6] Anonymous & Callback Functions

  1. Anonymous Function: Function without a name (often used inline).
  2. Callback Function: Passed as an argument to another function.
            function runExample6() 
            {
                // Anonymous Function with setTimeout
                setTimeout(function() 
                {
                console.log("Executed after 2 seconds (Anonymous Function)");
                }, 2000);

                // Callback Function
                function greetUser(name, callback) 
                {
                let msg = "Hello, " + name;
                console.log(msg);
                callback();
                return msg;
                }

                let result = greetUser("Alice", function() 
                {
                console.log("Callback executed!");
                });

                document.getElementById("output6").textContent = result + " (check console)";
            }

7] Variable Scope

  1. Global Scope: Accessible everywhere
  2. Local Scope: Accessible only inside functions
            function runExample7() 
            {
            let globalVar = "I am Global";

            function testScope() 
            {
                let localVar = "I am Local";
                console.log(globalVar); // Accessible
                console.log(localVar);  // Accessible
                return globalVar + " | " + localVar;
            }

            let output = testScope();
            document.getElementById("output7").textContent = output;

            console.log(globalVar); // Accessible
            // console.log(localVar); // ❌ Error
        }

8] Lexical Scoping & Closures

  1. Lexical Scope: Inner functions can access outer variables.
  2. Closure: Function remembers variables even after outer function ends
    function run8() 
    {
        function outer() 
        {
            let outerVar = "I am outside!";
            function inner() {
                return outerVar;
        }
    return inner;
    }

    const closureFunc = outer();
    document.getElementById("output8").textContent = closureFunc();

    console.log(closureFunc()); // "I am outside!"
    }

9] Default Parameters

  1. Lexical Scope: Inner functions can access outer variables.
  2. Closure: Function remembers variables even after outer function ends
      function run9() 
      {
        function greet(name = "Guest") 
        {
        return `Hello, ${name}!`;
        }

        let result = greet("Alice") + " | " + greet();
        document.getElementById("out9").textContent = result;

        console.log(greet("Alice")); // Hello, Alice!
        console.log(greet());        // Hello, Guest!
