1️⃣ What is a Function?

Concept: Two ways to define function.

  • A function is block of reusable code that performs a task.
  • Follow DRY principle -> Don't Repeat Yourself.
  • Can be called multiple times with different inputs.
    Syntax:    
                    function functionName() 
                    {
                        return result;
                    } 
                    
    //Web:    
                function run1()
                {
                    function  msg()
                    {
                        return "Here is Simple Message";
                    }
                document.getElementById("out1").textContent = msg();
                document.getElementById("cd").style.display = "block";
                console.log(msg());
                }

2️⃣ Function Declaration vs Function Expression

Concept:

  • Declaration: Hoisted → can be called before defined
  • Expression: Stored in variable, not hoisted
    Syntax:    
            // Declaration
                function add(a, b) { return a + b; }

            // Expression
                const addExp = function(a, b) { return a + b; };

                    
    //Web:    
                function runExample2() 
                {
                    function addDec(a, b) 
                    { 
                        return a + b; 
                    }
                    const addExp = function(a, b) 
                    {
                        return a + b; 
                    };

                    let result = "Declaration: " + addDec(5,3) +
                                " | Expression: " + addExp(5,3);

                    document.getElementById("output2").textContent = result;
                    console.log(result);
                }

3️⃣ Parameters vs Arguments

Concept: Two ways to define function.

  • Parameters: Variables listed in function definition
  • Arguments: Actual values passed during function call
    Syntax:    
                    function functionName() 
                    {
                        return result;
                    } 
                    
    //Web:    
                function runExample3() {
                function multiply(x, y) { return x * y; }
                let result = multiply(4, 5); // arguments = 4, 5

                document.getElementById("output3").textContent =
                "Multiply Result: " + result;

                console.log("Multiply:", result); // 20
            }

4️⃣ Return Statements

Concept: Two ways to define function.

  • return sends value back to caller.
  • Without return, function gives undefined.
    Syntax:    
                    function functionName(params) {
                    return value;
                    }
 
                    
    //Web:    
                function runExample4() {
                function square(num) { return num * num; }
                let result = square(6);

                document.getElementById("output4").textContent =
                "Square of 6: " + result;

                console.log("Square:", result); // 36
            }

5️⃣ Arrow Functions (ES6)

Concept:

  • Introduced in ES6 for cleaner syntax
  • Shorter than traditional functionsFollow DRY principle -> Don't Repeat Yourself.
  • Useful for callbacks and small tasks.
    Syntax:    
                    const functionName = (params) => expression; 
                    
    //Web:    
                function runExample5() {
                const greet = (name) => `Hello, ${name}!`;
                const square = x => x * x;

                let result = greet("Alice") + " | Square of 5: " + square(5);

                document.getElementById("output5").textContent = result;
                console.log(result);
            }

6️⃣ Anonymous & Callback Functions

Concept:

  • Anonymous Function: Function without a name (often used inline).
  • Callback Function: Passed as an argument to another function
    Syntax:    
                    function functionName() 
                    {
                        return result;
                    } 
                    
    //Web:    
                function runExample6() {
                // Anonymous Function with setTimeout
                setTimeout(function() {
                console.log("Executed after 2 seconds (Anonymous Function)");
                }, 2000);

                // Callback Function
                function greetUser(name, callback) {
                let msg = "Hello, " + name;
                console.log(msg);
                callback();
                return msg;
                }

                let result = greetUser("Alice", function() {
                console.log("Callback executed!");
                });

                document.getElementById("output6").textContent = result + " (check console)";
            }

7️⃣ Variable Scope

Concept: Two ways to define function.

  • Global Scope: Accessible everywhere.
  • Local Scope: Accessible only inside functions
    • \ 
    Syntax:    
                    function functionName() 
                    {
                        return result;
                    } 
                    
    //Web:    
                function runExample7() {
                    let globalVar = "I am Global";

                    function testScope() {
                    let localVar = "I am Local";
                    console.log(globalVar); // Accessible
                    console.log(localVar);  // Accessible
                    return globalVar + " | " + localVar;
                    }

                    let output = testScope();
                    document.getElementById("output7").textContent = output;

                    console.log(globalVar); // Accessible
                    // console.log(localVar); // ❌ Error
                }

8️⃣ Lexical Scoping & Closures

Concept: Two ways to define function.

  • Lexical Scope: Inner functions can access outer variables
  • Closure: Function remembers variables even after outer function ends
    Syntax:    
                    function functionName() 
                    {
                        return result;
                    } 
                    
    //Web:    
                function runExample8() {
                function outer() {
                let outerVar = "I am outside!";
                function inner() {
                    return outerVar;
                }
                return inner;
                }

                const closureFunc = outer();
                document.getElementById("output8").textContent = closureFunc();

                console.log(closureFunc()); // "I am outside!"
            }

Concept:

  • Allows assigning default values if no argument is passed.
    Syntax:

                    }

                console.log(greet("Alice")); // Hello, Alice!

                function runExample11() {

                function calculator(a, b, op) {

                            " | Counter: " + myCounter();

                console.log(result);
            }
