Bouncing Animation

Syntax

                @keyframes bounce {
                                    0%,
                                    20%,
                                    50%,
                                    80%,
                                    100% {
                                    transform: translateY(0);
                                    }
                                    40% {
                                    transform: translateY(-30px); /* jump up */
                                    }
                                    60% {
                                    transform: translateY(-15px); /* smaller bounce */
                                    }
                                }
animation: bounce 2s infinite;
